#include   <pthread.h> 
#include   <sys/types.h> 
#include   <unistd.h> 
#include   <stdio.h> 
#include   <stdlib.h> 

void   *thread(void*   m) 
{ 
                pthread_setcancelstate(   PTHREAD_CANCEL_ENABLE,   NULL);           //允许退出线程 
                pthread_setcanceltype(PTHREAD_CANCEL_ASYNCHRONOUS,   NULL);   //设置立即取消 
                unsigned   long     t,id; 
                t   =   *(unsigned   long   *)m; 
                while(1) 
                { 
                                printf( "thread   %d\n ",t);   //打印哪个线程 
                                sleep(3); 
                } 
}

{ 
                pthread_t   t; 
                pthread_attr_t   at;
